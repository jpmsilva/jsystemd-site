17  package com.github.jpmsilva.jsystemd;
18  
19  import java.util.Arrays;
20  import java.util.Map;
21  import java.util.Map.Entry;
22  import java.util.function.Function;
23  import java.util.function.Predicate;
24  import java.util.stream.Collectors;
25  import org.springframework.beans.BeansException;
26  import org.springframework.beans.factory.config.BeanDefinition;
27  import org.springframework.beans.factory.config.BeanPostProcessor;
28  import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
29  import org.springframework.core.annotation.Order;
30  
31  /**
32   * Implementation of {@link SystemdNotifyStatusProvider} that provides information regarding the bean creation progress of a bean factory.
33   *
34   * @author Joao Silva
35   * @see BeanPostProcessor
36   */
37  @Order(-4000)
38  public class SystemdNotifyApplicationContextStatusProvider implements SystemdNotifyStatusProvider, BeanPostProcessor {
39  
40    private final Systemd systemd;
41    private ConfigurableListableBeanFactory factory;
42    private Map<String, BeanDefinition> definitions;
43    private String status = "";
44  
45    /**
46     * Creates a new instance using the provided {@link Systemd} as the integration point.
47     *
48     * <p>This constructor automatically registers the created instance as a status provider in the {@link Systemd} passed as a parameter.
49     *
50     * @param systemd the {@link Systemd} to send status information to
51     */
52    SystemdNotifyApplicationContextStatusProvider(Systemd systemd) {
53      this.systemd = systemd;
54      this.systemd.addStatusProviders(this);
55    }
56  
57    /**
58     * Sets the bean factory once it's available.
59     *
60     * @param factory the bean factory to provide status information about
61     */
62    void setFactory(ConfigurableListableBeanFactory factory) {
63      this.factory = factory;
64    }
65  
66    /**
67     * {@inheritDoc}
68     */
69    @Override
70    public String status() {
71      return systemd.isReady() ? "" : status;
72    }
73  
74    /**
75     * {@inheritDoc}
76     */
77    @Override
78    public Object postProcessBeforeInitialization(Object bean, String beanName)
79        throws BeansException {
80      if (factory != null) {
81        ensureDefinitionsLoaded();
82        status = String.format("Creating bean %d of %d", getSingletonCount(), definitions.size());
83        systemd.extendTimeout();
84        systemd.updateStatus();
85      }
86      return bean;
87    }
88  
89    private void ensureDefinitionsLoaded() {
90      if (definitions == null) {
91        definitions = Arrays.stream(factory.getBeanDefinitionNames())
92            .filter(isSingleton(factory))
93            .collect(Collectors.toMap(Function.identity(), factory::getBeanDefinition));
94      }
95    }
96  
97    /**
98     * {@inheritDoc}
99     */
100   @Override
101   public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
102     return bean;
103   }
104 
105   private long getSingletonCount() {
106     return definitions.entrySet().stream()
107         .filter(containsSingleton(factory))
108         .map(Entry::getKey)
109         .count();
110   }
111 
112   private Predicate<? super String> isSingleton(ConfigurableListableBeanFactory factory) {
113     return beanName -> factory.getBeanDefinition(beanName).isSingleton();
114   }
115 
116   private Predicate<? super Map.Entry<String, BeanDefinition>> containsSingleton(ConfigurableListableBeanFactory factory) {
117     return entry -> factory.containsSingleton(entry.getKey());
118   }
119 }

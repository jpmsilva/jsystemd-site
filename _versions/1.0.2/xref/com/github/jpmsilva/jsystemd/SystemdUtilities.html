17  package com.github.jpmsilva.jsystemd;
18  
19  import static java.lang.invoke.MethodHandles.lookup;
20  import static org.slf4j.LoggerFactory.getLogger;
21  
22  import org.slf4j.Logger;
23  
24  /**
25   * Generic utilities for the systemd library. Not meant to be used directly or instantiated.
26   *
27   * @author Joao Silva
28   */
29  abstract class SystemdUtilities {
30  
31    private SystemdUtilities() {
32    }
33  
34    private static final Logger logger = getLogger(lookup().lookupClass());
35  
36    private static final String notifySocket = System.getenv("NOTIFY_SOCKET");
37    private static final String[] implClasses = new String[]{
38        "com.github.jpmsilva.jsystemd.SystemdNotifyNative",
39        "com.github.jpmsilva.jsystemd.SystemdNotifyProcess"
40    };
41    private static final SystemdNotify SYSTEMD_NOTIFY = initSystemdNotify();
42  
43    /**
44     * Logs information regarding the status of the integration with systemd. Meant to be used once the application has done sufficient work to initialize
45     * logging.
46     *
47     * <p>Only logs information when running under Linux.
48     */
49    static void logSystemdStatus() {
50      if (isLinux()) {
51        logger.info("Chosen systemd notify library: \"" + SYSTEMD_NOTIFY
52            + "\", OS name: \"" + osName() + "\""
53            + ", notify socket: \"" + notifySocket() + "\"");
54      }
55    }
56  
57    /**
58     * Allows determining if the process is running under systemd.
59     *
60     * <p>A process is said to be running under systemd if <ol> <li>the operating system name contains {@code linux}.</li> <li>an environment property {@code
61     * NOTIFY_SOCKET} exists</li> <li>an implementation of {@link SystemdNotify} was able to initialize itself</li> </ol>
62     *
63     * @return {@code true} if the process is running under systemd
64     * @see SystemdNotify#usable()
65     */
66    static boolean isUnderSystemd() {
67      return isLinux() && hasNotifySocket();
68    }
69  
70    /**
71     * Allows determining the current OS name.
72     *
73     * @return the contents of the system property {@code os.name}
74     */
75    static String osName() {
76      return System.getProperty("os.name");
77    }
78  
79    /**
80     * Allows determining the current systemd notify socket.
81     *
82     * @return the contents of the environment property {@code NOTIFY_SOCKET}
83     */
84    static String notifySocket() {
85      return notifySocket;
86    }
87  
88    /**
89     * Allows determining the current {@link SystemdNotify} implementation.
90     *
91     * @return the current {@link SystemdNotify} implementation
92     */
93    static SystemdNotify getSystemdNotify() {
94      return SYSTEMD_NOTIFY;
95    }
96  
97    private static SystemdNotify initSystemdNotify() {
98      SystemdNotify systemdNotify = new SystemdNotifyNoop();
99      if (isUnderSystemd()) {
100       for (String implClass : implClasses) {
101         SystemdNotify impl = getImpl(implClass);
102         if (impl != null) {
103           systemdNotify = impl;
104           break;
105         }
106       }
107     }
108     return systemdNotify;
109   }
110 
111   @SuppressWarnings("checkstyle:EmptyCatchBlock")
112   private static SystemdNotify getImpl(String implClass) {
113     try {
114       SystemdNotify systemdNotify = (SystemdNotify) Class.forName(implClass).newInstance();
115       if (systemdNotify.usable()) {
116         return systemdNotify;
117       }
118     } catch (Exception ignored) {
119     }
120     return null;
121   }
122 
123   private static boolean isLinux() {
124     return osName().toLowerCase().startsWith("linux");
125   }
126 
127   private static boolean hasNotifySocket() {
128     return null != notifySocket;
129   }
130 }
